# Conan dependencies for the native Windows (MinGW) build.
#
# Same versions as the top-level conanfile.txt, but WITHOUT Boost: on MinGW
# Boost is built from source (the Conan Boost recipe runs `b2`, which hits a
# MinGW static-archive bug — see WINDOWS.md). zlib and bzip2 are listed
# explicitly here because on Linux/macOS they come in transitively via Boost.
# Iostreams, whereas the Windows build needs them for the spatialjoin
# dependency. Keep the icu/openssl/zstd versions in sync with conanfile.txt.
[requires]
icu/76.1
openssl/3.1.1
zstd/1.5.5
zlib/1.3.2
bzip2/1.0.8

[generators]
CMakeDeps
CMakeToolchain
